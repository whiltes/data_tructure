####冒泡排序

```
#include <stdio.h>

void bubble_sort(int arr[], int sz)
{
	int i = 0;
	int j = 0;

	for(i=0; i<sz-1; i++)
	{
		for(j=0; j<sz-1; j++)
		{
			if(arr[j]>arr[j+1])
			{
				int tmp = arr[j];
				arr[j] = arr[j+1];
				arr[j+1] = tmp;
			}
		}
	}
}

int main()
{
	int arr[10] = {0};
	int i = 0;
	int j = 0;
	int sz = sizeof(arr)/sizeof(arr[0]);

	for(i=0; i<10; i++)
	{
		scanf("%d", &arr[i]);
	}

	bubble_sort(arr, sz);
	for(i=0; i<10; i++)
	{
		printf("%d ", arr[i]);
	}
	putchar('\n');

	return 0;
}
```
